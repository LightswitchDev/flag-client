## Install

```bash
npm install --save lightswitch-client
```

## Usage

```tsx
import * as React from 'react'

import { useMyHook } from 'lightswitch-client'

const Example = () => {
  const example = useMyHook()
  return (
    <div>
      {example}
    </div>
  )
}
```
